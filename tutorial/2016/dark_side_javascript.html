typeof null

Это оффициально признанный баг 5, не фиксищейся для поддержки совместимости со старым кодом 12. typeof null возвращает "object", однако null не является объектом 5, и что хуже, не ведёт себя как объект.

typeof null     // "object"

var z = null;   // null
z.prop = 1      // TypeError: z is null

Если мы заглянем в код библиотеки underscore.js 4, то увидим такую функцию как isObject(), как раз правильно проверяющую на тип объекта.

// версия 1.6.0 (2014)
_.isObject = function(obj) {
    return obj === Object(obj);
};

// версия 1.8.3 (2015)
// соответствует lodash
_.isObject = function(obj) {
    var type = typeof obj;
    return type === 'function' || type === 'object' && !!obj;
};

_.isObject(null)    // false

А также функцию isNull() проверяющую на соответствие null (тоже самое в lodash):

_.isNull = function(obj) {
    return obj === null;
};

_.isNull(null)      // true
_.isNull({})        // false
_.isNull(undefined) // false
_.isNull(NaN)       // false

Что есть NaN?

Возможно вы не знаете, но NaN является числом 3 (и не важно, что NaN акроним от Not a Number), просто это число никогда не равно самому себе 146.

typeof NaN  // "number"

NaN === NaN // false

В underscore.js есть пример проверки на NaN 4 (тоже самое в lodash):

_.isNaN = function(obj) {
    return toString.call(obj) === '[object Number]' && obj !== +obj;
};

_.isNaN(NaN)    // true

Есть также встроенная функция isNaN(), но она работает некорректно 6:

isNaN(NaN)          // true
isNaN(undefined)    // true
isNaN({})           // true
isNaN(new Date().toString())   // true
isNaN("words")      // true

В новом стандарте ECMAScript 6 (Harmony) предложена новая функция Number.isNaN() избавленная от этих недостатков. Вот её полифилл 7:

Number.isNaN = Number.isNaN || function(value) {
  return typeof value === 'number' && isNaN(value);
}

Преобразование типов

[]+[]   // ""
[]+{}   // "[object Object]"
{}+[]   // 0
{}+{}   // NaN

[]-[]   // 0
[]-{}   // NaN
{}-[]   // -0
{}-{}   // NaN

Там где вначале строки стоит { -- там левая фигурная скобка рассматривается как начало блока кода.
